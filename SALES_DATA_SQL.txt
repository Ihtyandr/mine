CREATE TABLE SALES_DATA 
(
  YEAR VARCHAR2(20 BYTE)  
, QUARTER VARCHAR2(20 BYTE) 
, MONTH VARCHAR2(20 BYTE) 
, FULL_DATE DATE 
, CATEGORY VARCHAR2(50 BYTE) 
, SUB_CATEGORY VARCHAR2(20 BYTE) 
, CUSTOMER_AGE_GROUP VARCHAR2(20 BYTE) 
, CUSTOMER_GENDER VARCHAR2(20 BYTE) 
, CUSTOMER_COUNTRY VARCHAR2(20 BYTE) 
, SALES_TERRITORY VARCHAR2(20 BYTE) 
, REVENUE NUMBER
, NUMBER_OF_ORDERS NUMBER 
);
